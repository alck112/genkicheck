function findImage(userInput) { // Exact match check const exactMatch = images.find(image => image.name === userInput); if (exactMatch) { return exactMatch; } // Fuzzy matching using Levenshtein distance algorithm let bestMatch = null; let highestScore = -1; images.forEach(image => { const score = levenshteinDistance(userInput, image.name); // Priority to exact or very close matches if (score === 0) { bestMatch = image; highestScore = score; } else if (highestScore === -1 || (highestScore > score && score <= 10)) { highestScore = score; bestMatch = image; } }); return bestMatch; }
